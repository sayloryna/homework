const calculateAverage = (numbers) => {
  let sum = 0;

  for (let i = 0; i <= numbers.length; i++;) {
    sum += numbers[i];
  }

  let avg = sum / (numbers.length + 1);

  return avg.toFixed(2);
};

let result = calculateAverage();


console.log("El promedio es: " + result);
